EXPECTED FLOW:
==============
1. PDO Payment Created
   - Database: status = "pending", pdo_approval_status = "pending"

   - API Call: POST /api/payments/{id}/verify
   - Backend Updates: status = "verified"

VERIFICATION SCRIPT:
===================

Run this in Laravel Tinker to check everything:

use App\Models\Payment;
use App\Models\Sale;

echo "=== PDO PAYMENT SYSTEM CHECK ===" . PHP_EOL;
echo PHP_EOL;

// 1. Check schema
echo "1. Checking database schema..." . PHP_EOL;
$hasColumn = Schema::hasColumn('payments', 'pdo_approval_status');
echo "   - pdo_approval_status column exists: " . ($hasColumn ? 'YES ✓' : 'NO ✗') . PHP_EOL;
echo PHP_EOL;

// 2. Count PDO payments
echo "2. Counting PDO payments..." . PHP_EOL;
$totalPDOs = Payment::where('payment_method', 'pdo')->count();
echo "   - Total PDO payments: $totalPDOs" . PHP_EOL;

if ($totalPDOs > 0) {
    $pending = Payment::where('payment_method', 'pdo')->where('pdo_approval_status', 'pending')->count();
    $approved = Payment::where('payment_method', 'pdo')->where('pdo_approval_status', 'approved')->count();
    $verified = Payment::where('payment_method', 'pdo')->where('status', 'verified')->count();
    
    echo "   - Pending approval: $pending" . PHP_EOL;
    echo "   - Approved (awaiting payment): $approved" . PHP_EOL;
    echo "   - Verified (completed): $verified" . PHP_EOL;
    echo PHP_EOL;
    
    // 3. Show sample PDO
    echo "3. Sample PDO payment:" . PHP_EOL;
    $sample = Payment::where('payment_method', 'pdo')->first();
    if ($sample) {
        echo "   - ID: {$sample->id}" . PHP_EOL;
        echo "   - Order: {$sample->sale->transaction_id}" . PHP_EOL;
        echo "   - Amount: ₱" . number_format($sample->amount, 2) . PHP_EOL;
        echo "   - Check Number: {$sample->pdo_check_number}" . PHP_EOL;
        echo "   - Status: {$sample->status}" . PHP_EOL;
        echo "   - PDO Approval: " . ($sample->pdo_approval_status ?? 'NULL') . PHP_EOL;
    }
} else {
    echo "   - No PDO payments found. Create one via POS to test." . PHP_EOL;
}

echo PHP_EOL;
echo "=== END CHECK ===" . PHP_EOL;
